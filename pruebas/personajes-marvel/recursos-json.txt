import com.google.gson.Gson;
import com.google.gson.GsonBuilder;
import com.google.gson.reflect.TypeToken;
import java.io.File;
import java.io.FileWriter;
import java.io.IOException;
import java.lang.reflect.Type;
import java.nio.file.Files;
import java.nio.file.Paths;
import java.util.ArrayList;
import java.util.List;
import java.util.Scanner;

public class Main {

    private static final String JSON_FILE_PATH = "datos.json";

    public static void main(String[] args) {
        List<Persona> personajes = cargarDatos();
        Scanner scanner = new Scanner(System.in);

        while (true) {
            System.out.println("Opciones:");
            System.out.println("1. Listar personajes");
            System.out.println("2. Agregar personaje");
            System.out.println("3. Actualizar personaje");
            System.out.println("4. Eliminar personaje");
            System.out.println("5. Salir");
            System.out.print("Seleccione una opción: ");
            int opcion = scanner.nextInt();
            scanner.nextLine();

            switch (opcion) {
                case 1:
                    listarPersonas(personajes);
                    break;
                case 2:
                    agregarPersona(scanner, personajes);
                    break;
                case 3:
                    actualizarPersona(scanner, personajes);
                    break;
                case 4:
                    eliminarPersona(scanner, personajes);
                    break;
                case 5:
                    guardarDatos(personajes);
                    System.out.println("Saliendo...");
                    return;
                default:
                    System.out.println("Opción inválida");
            }
        }
    }

    private static List<Persona> cargarDatos() {
        try {
            String json = new String(Files.readAllBytes(Paths.get(JSON_FILE_PATH)));
            Type listType = new TypeToken<ArrayList<Persona>>(){}.getType();
            return new Gson().fromJson(json, listType);
        } catch (IOException e) {
            System.err.println("Error al cargar datos: " + e.getMessage());
            return new ArrayList<>();
        }
    }

    private static void guardarDatos(List<Persona> personajes) {
        try (FileWriter writer = new FileWriter(JSON_FILE_PATH)) {
            new GsonBuilder().setPrettyPrinting().create().toJson(personajes, writer);
        } catch (IOException e) {
            System.err.println("Error al guardar datos: " + e.getMessage());
        }
    }

    private static void listarPersonas(List<Persona> personajes) {
        if (personajes.isEmpty()) {
            System.out.println("No hay personajes registradas");
        } else {
            System.out.println("Listado de personajes:");
            for (Persona personaje : personajes) {
                System.out.println(personaje);
            }
        }
    }

    private static void agregarPersona(Scanner scanner, List<Persona> personajes) {
        System.out.print("Ingrese el nombre: ");
        String nombre = scanner.nextLine();
        System.out.print("Ingrese la edad: ");
        int edad = scanner.nextInt();
        scanner.nextLine();
        System.out.print("Ingrese el correo: ");
        String correo = scanner.nextLine();

        personajes.add(new Persona(nombre, edad, correo));
        System.out.println("Persona agregada exitosamente");
    }

    private static void actualizarPersona(Scanner scanner, List<Persona> personajes) {
        System.out.print("Ingrese el ID de la personaje a actualizar: ");
        int id = scanner.nextInt();
        scanner.nextLine();

        for (Persona personaje : personajes) {
            if (personaje.getId() == id) {
                System.out.print("Ingrese el nuevo nombre: ");
                String nombre = scanner.nextLine();
                System.out.print("Ingrese la nueva edad: ");
                int edad = scanner.nextInt();
                scanner.nextLine();
                System.out.print("Ingrese el nuevo correo: ");
                String correo = scanner.nextLine();

                personaje.setNombre(nombre);
                personaje.setEdad(edad);
                personaje.setCorreo(correo);
                System.out.println("Persona actualizada exitosamente");
                return;
            }
        }
        System.out.println("No se encontró ninguna personaje con el ID proporcionado");
    }

    private static void eliminarPersona(Scanner scanner, List<Persona> personajes) {
        System.out.print("Ingrese el ID de la personaje a eliminar: ");
        int id = scanner.nextInt();
        scanner.nextLine();

        for (Persona personaje : personajes) {
            if (personaje.getId() == id) {
                personajes.remove(personaje);
                System.out.println("Persona eliminada exitosamente");
                return;
            }
        }
        System.out.println("No se encontró ninguna personaje con el ID proporcionado");
    }
}
